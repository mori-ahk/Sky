factorial(integer x) : integer
    do
        if (x == 1)
        then return(1);
        else
           return(factorial(x-1) * x);
        ;
    end

add(integer x) : integer
    do
        return (x + 2);
    end

print(integer x, integer y): integer
    do
        return (x + y);
    end

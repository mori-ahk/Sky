/* sort the array */
bubbleSort(integer arr[], integer size) : void
  local
    integer n;
    integer i;
    integer j;
    integer temp;
  do
    n = size;
    i = 0;
    j = 0;
    temp = 0;
    while (i < n-1)
      do
        while (j < n-i-1)
          do
            if (arr[j] > arr[j+1])
              then
                do
                  // swap temp and arr[i]
                  temp = arr[j];
                  arr[j] = arr[j+1];
                  arr[j+1] = temp;
                end
              else
	        ;
            j = j+1;
          end;
        i = i+1;
      end;
  end

/* Print the array */
printArray(integer arr[], integer size) : void
  local
    integer n;
    integer i;
  do
    n = size;
    i = 0;
    while (i<n)
      do
        write(arr[i]);
        i = i+1;
      end;
  end

main
  local
    integer arr[7];
  do
    arr[0] = 64;
    arr[1] = 34;
    arr[2] = 25;
    arr[3] = 12;
    arr[4] = 22;
    arr[5] = 11;
    arr[6] = 90;
    printarray(arr, 7);
    bubbleSort(arr, 7);
